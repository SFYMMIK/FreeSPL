#include "utils.h"

void example_usage() {
    char* str = string_duplicate("Hello, World!");
    char* num_str = int_to_string(42);
    char* combined = string_concat(str, num_str);

    printf("%s\n", combined);  // Outputs: Hello, World!42

    free(str);
    free(num_str);
    free(combined);
}
